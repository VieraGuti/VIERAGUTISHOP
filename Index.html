Generador de Comandos

function generateCommand() { const displayName = document.getElementById('display_name').value; const shortName = document.getElementById('item_list').value; const quantity = document.getElementById('quantity').value; const cost = document.getElementById('cost').value; const server = document.getElementById('server').value; if (!displayName || shortName === "none" || !quantity || !cost || !server) { alert('Por favor, completa todos los campos.'); return; } // Formato del comando const command = `/additem display_name: "${displayName}" short_name: "${shortName}" quantity: ${quantity} cost: ${cost} server: ${server}`; commands.push(command); localStorage.setItem('commands', JSON.stringify(commands)); displayCommands(); } function displayCommands() { const commandList = document.getElementById('command_list'); commandList.innerHTML = ''; commands.forEach((cmd, index) => { const commandItem = document.createElement('div'); commandItem.className = 'command-item'; commandItem.innerHTML = ` ${cmd}
`; commandList.appendChild(commandItem); }); } function copyCommand(index) { const tempInput = document.createElement('input'); tempInput.value = commands[index]; document.body.appendChild(tempInput); tempInput.select(); document.execCommand('copy'); document.body.removeChild(tempInput); alert('Comando copiado: ' + commands[index]); // Añadir mensaje de "ya copiado" const commandItem = document.querySelectorAll('.command-item')[index]; const copiedMessage = document.createElement('span'); copiedMessage.className = 'copied-message'; copiedMessage.textContent = ' - Ya copiado'; commandItem.appendChild(copiedMessage); } function deleteCommand(index) { commands.splice(index, 1); localStorage.setItem('commands', JSON.stringify(commands)); displayCommands(); } // Cargar los comandos guardados al iniciar la página window.onload = function() { const savedCommands = localStorage.getItem('commands'); if (savedCommands) { commands.push(...JSON.parse(savedCommands)); displayCommands(); } };
